130 template <typename T>
131 SymbolicValue BinaryVisitor<T>::Dispatch()
132 {
133 #define DISPATCH(value) \
134  switch (rhs_.GetKind()) { \
135  case SymbolicValue::Kind::SCALAR: \
136  return Visit(value, Scalar{}); \
137  case SymbolicValue::Kind::LOWER_BOUNDED_INTEGER: \
138  return Visit(value, LowerBoundedInteger{rhs_.GetInteger()}); \
139  case SymbolicValue::Kind::UNDEFINED: \
140  return Visit(value, Undefined{}); \
141  case SymbolicValue::Kind::INTEGER: \
142  return Visit(value, rhs_.GetInteger()); \
143  case SymbolicValue::Kind::FLOAT: \
144  return Visit(value, rhs_.GetFloat()); \
145  case SymbolicValue::Kind::POINTER: \
146  return Visit(value, Pointer{rhs_.GetPointer()}); \
147  case SymbolicValue::Kind::NULLABLE: \
148  return Visit(value, Nullable{rhs_.GetPointer()}); \
149  case SymbolicValue::Kind::VALUE: \
150  return Visit(value, Value{rhs_.GetPointer()}); \
151  } \
152  llvm_unreachable("invalid rhs kind");
153 
154  switch (lhs_.GetKind()) {
155  case SymbolicValue::Kind::SCALAR:
156  DISPATCH(Scalar{});
157  case SymbolicValue::Kind::LOWER_BOUNDED_INTEGER:
158  DISPATCH(LowerBoundedInteger{lhs_.GetInteger()});
159  case SymbolicValue::Kind::INTEGER:
160  DISPATCH(lhs_.GetInteger());
161  case SymbolicValue::Kind::FLOAT:
162  DISPATCH(lhs_.GetFloat());
163  case SymbolicValue::Kind::POINTER:
164  DISPATCH(Pointer{lhs_.GetPointer()});
165  case SymbolicValue::Kind::VALUE:
166  DISPATCH(Value{lhs_.GetPointer()});
167  case SymbolicValue::Kind::NULLABLE:
168  DISPATCH(Nullable{lhs_.GetPointer()});
169  case SymbolicValue::Kind::UNDEFINED:
170  DISPATCH(Undefined{});
171  }
172  llvm_unreachable("invalid lhs kind");
173 }
