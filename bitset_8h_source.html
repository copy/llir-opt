21 template<typename T, unsigned N = 8>
22 class BitSet final {
23 private:
25  static constexpr uint64_t kBitsInBucket = sizeof(uint64_t) * CHAR_BIT;
27  static constexpr uint64_t kBitsInChunk = kBitsInBucket * N;

30  struct Node {
31  public:
32  Node()
33  {
34  for (unsigned i = 0; i < N; ++i) {
35  arr[i] = 0ull;
36  }
37  }
38 
39  bool Insert(unsigned bit)
40  {
41  const uint64_t bucket = bit / kBitsInBucket;
42  const uint64_t mask = 1ull << (bit - bucket * kBitsInBucket);
43  const uint64_t val = arr[bucket];
44  const bool inserted = !(val & mask);
45  arr[bucket] = val | mask;
46  return inserted;
47  }
48 
49  bool Contains(unsigned bit) const
50  {
51  uint64_t bucket = bit / kBitsInBucket;
52  const uint64_t mask = 1ull << (bit - bucket * kBitsInBucket);
53  return arr[bucket] & mask;
54  }
55 
56  bool Erase(unsigned bit)
57  {
58  const uint64_t bucket = bit / kBitsInBucket;
59 
60  arr[bucket] &= ~(1ull << (bit - bucket * kBitsInBucket));
61 
62  for (unsigned i = 0; i < N; ++i) {
63  if (arr[i] != 0ull) {
64  return false;
65  }
66  }
67  return true;
68  }
69 
70  size_t Size() const
71  {
72  size_t size = 0;
73  for (unsigned i = 0; i < N; ++i) {
74  size += __builtin_popcountll(arr[i]);
75  }
76  return size;
77  }
78 
79  bool operator==(const Node &that) const
80  {
81  for (unsigned i = 0; i < N; ++i) {
82  if (arr[i] != that.arr[i]) {
83  return false;
84  }
85  }
86  return true;
87  }
88 
89  unsigned Union(const Node &that)
90  {
91  unsigned changed = 0;
92  for (unsigned i = 0; i < N; ++i) {
93  uint64_t old = arr[i];
94  arr[i] |= that.arr[i];
95  changed += __builtin_popcountll(old ^ arr[i]);
96  }
97  return changed;
98  }

102  };

303  void Erase(const ID<T> &item)
304  {
305  if (item == first_ && item == last_) {
306  first_ = std::numeric_limits<uint32_t>::max();
307  last_ = std::numeric_limits<uint32_t>::min();
308  } else if (item == first_) {
309  first_ = *++begin();
310  } else if (item == last_) {
311  last_ = *++rbegin();
312  }
313 
314  auto &node = nodes_[item / kBitsInChunk];
315  if (node.Erase(item - (item / kBitsInChunk) * kBitsInChunk)) {
316  nodes_.erase(item / kBitsInChunk);
317  }
318  }

327  if (it == nodes_.end()) {
